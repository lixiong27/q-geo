├── index.jsx                    # GEO 分析入口（Tab 导航）
├── TemplateManage.jsx           # 模板管理页面
├── ResultReport.jsx             # 结果报表页面
├── components/
│   ├── TemplateForm.jsx         # 模板表单组件
│   ├── GroupConfig.jsx          # 热词组配置组件
│   ├── ResultDetail.jsx         # 结果详情组件

    └── geo.js                   # GEO API（新增接口）

新增 API 接口

接口 方法 用途

/api/geoTemplate/toggle POST 启用/停用模板

/api/geoResult/list GET 结果列表查询
/api/geoResult/detail GET 结果详情
